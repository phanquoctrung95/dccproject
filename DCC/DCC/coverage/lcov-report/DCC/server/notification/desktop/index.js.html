function binarySearch(key, first, last) {
    var mid;
    key = key.toUpperCase();
    while (first <= last) {
        mid = Math.floor((last + first) / 2);
        if (onlineUsers[mid].email) {
            if (key == onlineUsers[mid].email.toUpperCase())
                return mid;
            if (key > onlineUsers[mid].email.toUpperCase())
                first = mid + 1;
            if (key < onlineUsers[mid].email.toUpperCase())
                last = mid - 1;
        }
    }
    return -1;
}
